def exp_for_level(level)
  lv = level.to_f
  basis = @exp_params[0].to_f
  extra = @exp_params[1].to_f
  acc_a = @exp_params[2].to_f
  acc_b = @exp_params[3].to_f
  return (basis*((lv-1)**(0.9+acc_a/250))*lv*(lv+1)/
    (6+lv**2/50/acc_b)+(lv-1)*extra).round.to_i
end
