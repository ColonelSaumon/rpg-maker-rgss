def initialize(width, height)
  @display_name = ''
  @tileset_id = 1
  @width = width
  @height = height
  @scroll_type = 0
  @specify_battleback = false
  @battleback_floor_name = ''
  @battleback_wall_name = ''
  @autoplay_bgm = false
  @bgm = RPG::BGM.new
  @autoplay_bgs = false
  @bgs = RPG::BGS.new('', 80)
  @disable_dashing = false
  @encounter_list = []
  @encounter_step = 30
  @parallax_name = ''
  @parallax_loop_x = false
  @parallax_loop_y = false
  @parallax_sx = 0
  @parallax_sy = 0
  @parallax_show = false
  @note = ''
  @data = Table.new(width, height, 4)
  @events = {}
end
